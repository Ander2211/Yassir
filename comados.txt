# --- Verificar conectividad con el servidor Fedora ---
ping 172.16.0.10

# --- Detener y deshabilitar systemd-resolved ---
sudo systemctl disable systemd-resolved --now

# --- Eliminar resolv.conf viejo ---
sudo rm /etc/resolv.conf

# --- Crear resolv.conf nuevo apuntando a Fedora ---
echo "nameserver 172.16.0.10" | sudo tee /etc/resolv.conf

# --- Probar resolución contra Fedora ---
dig @172.16.0.10 semita.sv
dig @172.16.0.10 www.semita.sv
dig -x 172.16.0.10 @172.16.0.10

# --- Probar resolución normal (usando resolv.conf) ---
dig semita.sv
dig www.semita.sv
